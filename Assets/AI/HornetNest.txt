tree("Root")
	fallback
		tree("Idle")
		tree("Combat")
		tree("Lost Target")
tree("Idle")
	sequence
		not isInCombat
tree("Combat")
	while 
		isInCombat
		repeat
			sequence
				Wait(0.1)
				CheckInformation()
				tree("Actions")	
				Wait(0.2)
				checkEndTurn()
tree("Actions")
	fallback
		CallAttackItems()


tree("Lost Target")
	sequence
		not isInCombat
		Wait(0.1)
		MakeQuestionMark()
		SetActionPoints(0)
		Wait(0.5)
		checkEndTurn()
