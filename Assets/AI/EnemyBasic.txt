tree("Root")
	fallback
		tree("Idle")
		tree("Combat")
		tree("Lost Target")
tree("Idle")
	sequence
		not isInCombat
		WalkRandomly()
tree("Combat")
	while 
		isInCombat
		repeat
			sequence
				Wait(0.1)
				CheckInformation()
				tree("Actions")	
				Wait(0.2)
				checkEndTurn()
tree("Actions")
	fallback
		Attack()
		MoveToTarget()
		AdvanceToTarget()


tree("Lost Target")
	sequence
		not isInCombat
		Wait(0.1)
		MakeQuestionMark()
		SetActionPoints(0)
		Wait(0.5)
		checkEndTurn()
