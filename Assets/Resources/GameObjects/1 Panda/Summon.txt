tree("Root")
	fallback
		not isCurrentTurnCharacter
		tree("Combat")
tree("Combat")
	repeat
		sequence
			isCurrentTurnCharacter
			WaitForStack()
			CheckInformation()
			tree("Actions")
			StartStack()
			WaitForStack()
			checkEndTurn()
tree("Actions")
	fallback
		Attack()
		not isHealthAboveZero
		tree("Move")
	

tree("Move")
	sequence
		fallback
			MoveToTarget()
			AdvanceToTarget()
		Wait(0.3)
